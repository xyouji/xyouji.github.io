    <!-- JavaScript to calculate and display the date -->
    <script>
        function calculateDate() {
            const startDate = new Date('1983-07-16');
            const currentDate = new Date();
            const diffTime = Math.abs(currentDate - startDate);
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            return diffDays.toString(36).toUpperCase();
        }

        document.getElementById('date').innerText = `Today is day ${calculateDate()}`;
    </script>
